>功能概述                                                         
  编写服务器端， 服务器端启动， 然后启动客户端， 通过客户端可以输入以下命令进行服务器上的文件查看：
  cd 进入对应目录
  ls 列出相应目录文件
  puts 将本地文件上传至服务器
  gets 文件名 下载服务器文件到本地
  remove 删除服务器上文件
  pwd 显示目前所在路径
  其他命令不响应
  支持密码验证、断点续传、数据库连接、多点下载；

>数据结构
  typedef struct tag_node{ //任务节点
  	int new_fd; //accept客户端请求生成的描述符
  	struct tag_node* pNext;
  }node_t, *pnode_t ;
  
  typedef struct{ //任务队列
  	pnode_t que_head, que_tail;
  	int que_capacity; //队列最大容量
  	int size; //队列当前大小
  	pthread_mutex_t que_mutex; //任务队列锁,入队出队都要加解锁
  }que_t, *pque_t;
  
  typedef void* (*thread_func_t)(void* p);
  typedef struct{ //线程池数据结构
  	pthread_t* pth_id; //线程id的起始地址
  	int pthread_num; //创建的线程数目
  	que_t que; //任务队列
  	thread_func_t threadfunc; //线程函数
  	pthread_cond_t cond;
  	short start_flag; //线程是否启动标志
  }factory, *pfac;
  
  typedef struct{//多点下载时记录相关信息                                                                                         
  	char ip[128];//存储不同服务器的地址，用于多点下载时连接不同的服务器
  	char port[128];
  	char filename[128];
  	int offset; //文件偏移值，记录多点下载时的起始地址；
  	int down_size; //需要下载的大小
  	int userid;
  }down, *pdown;
  
  typedef struct{//文件传输时的协议
  	int len; //控制数据，记录buf的大小
  	char buf[1000]; //文件内容
  }train;
  
  typedef struct{
  	char filename[128];//文件名
  	int sfd;//客户端与服务器连接的描述符
  }net, *pnet;

>数据库
  1.设计思想	
采用虚拟目录，在服务器上找一个目录，作为文件池，所有用户的所有文件都放在该目录下，且只存文件、每个文件只存一次，没有目录，这样可以节省存储空间，防止相同的文件重复存储（通过MD5码确定文件的唯一性）。
